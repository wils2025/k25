<body>
    <h1>Demostración de una Pila (LIFO)</h1>

    <button onclick="mostrarPila()">Mostrar Pila</button>
    <div id="resultado"></div>

    <script>
        // Creamos la pila usando un arreglo
        let pila = [];

        // Apilamos los números
        pila.push(8);
        pila.push(9);
        pila.push(12);

        // Función para mostrar la pila en LIFO
        function mostrarPila() {
            let resultado = "";
            while (pila.length > 0) {
                // pop() saca el último elemento agregado (LIFO)
                resultado += pila.pop() + "<br>";
            }
            document.getElementById("resultado").innerHTML = resultado;
        }
    </script>
